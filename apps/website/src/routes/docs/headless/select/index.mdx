# Select

This element is used to create a drop-down list, it's often used in a form, to collect user input.

## Building blocks

    ```tsx
    import { component$ } from '@builder.io/qwik';
    import { SelectRoot, SelectLabel, SelectTrigger, SelectValue, SelectMarker, SelectListBox, SelectOption, SelectGroup } from '@qwik-ui/headless';

    export default component$(() => (
      <SelectRoot>
        <SelectLabel>Content</SelectLabel>
        <SelectTrigger>
          <SelectValue>Content</SelectValue>
          <SelectMarker />
        </SelectTrigger>
        <SelectListBox>
          <SelectGroup>
            <SelectLabel>Options</SelectLabel>
            <SelectOption>Value</SelectOption>
            <SelectOption>Value</SelectOption>
            <SelectOption>Value</SelectOption>
          </SelectGroup>
        </SelectListBox>
      </SelectRoot>
    ));
    ```
