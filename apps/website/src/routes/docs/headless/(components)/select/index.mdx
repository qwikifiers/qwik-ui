# Select

This element is used to create a drop-down list, it's often used in a form, to collect user input.

{' '}

<Example01>
  ```tsx
  <SelectRoot class="dark:bg-gray-700">
    <SelectLabel class="ml-2 font-semibold text-white">Qwik Fruits</SelectLabel>
    <SelectTrigger class="group peer flex items-center justify-between rounded-md border-[1px] border-slate-200 bg-slate-100 p-4 px-8 dark:border-gray-600 dark:bg-gray-700">
      <SelectValue
        placeholder="Select a fruit! 🍹"
        class="text-gray-700 dark:text-white"
      />
      <SelectMarker class="h-6 w-6">
        <svg
          xmlns="http://www.w3.org/2000/svg"
          viewBox="0 0 24 24"
          fill="none"
          stroke-width="2"
          class="stroke-gray-700 transition-transform duration-[450ms] group-aria-expanded:-rotate-180 dark:stroke-white"
          stroke-linecap="round"
          stroke-linejoin="round"
        >
          <polyline points="6 9 12 15 18 9"></polyline>
        </svg>
      </SelectMarker>
    </SelectTrigger>
    <SelectListBox class="mt-2 rounded-md border-[1px] border-slate-200 bg-slate-100 dark:border-gray-600 dark:bg-gray-700">
      <SelectOption value="🚀 Qwik" class="p-4">
        🚀 Qwik
      </SelectOption>
      <SelectGroup class="p-4 ">
        <SelectLabel class="p-4">Fruits</SelectLabel>
        {[
          { value: '🍎 Apple', disabled: false },
          { value: '🍌 Banana', disabled: false },
          { value: '🍒 Cherry', disabled: false },
          { value: '🐲 Dragonfruit', disabled: true },
        ].map((option) => {
          return (
            <SelectOption
              key={option.value}
              value={option.value}
              disabled={option.disabled}
              class="rounded-sm p-4 hover:bg-slate-200 aria-disabled:cursor-not-allowed aria-disabled:text-red-500 dark:hover:bg-gray-600"
            >
              {option.value}
            </SelectOption>
          );
        })}
      </SelectGroup>
    </SelectListBox>
  </SelectRoot>
  ```
</Example01>

## Building blocks

  <CodeExample>
    ```tsx
    import { component$ } from '@builder.io/qwik';
    import { SelectRoot, SelectLabel, SelectTrigger, SelectValue, SelectMarker, SelectListBox, SelectOption, SelectGroup } from '@qwik-ui/headless';

    export default component$(() => (
      <SelectRoot>
        <SelectLabel>Content</SelectLabel>
        <SelectTrigger>
          <SelectValue>Content</SelectValue>
          <SelectMarker />
        </SelectTrigger>
        <SelectListBox>
          <SelectGroup>
            <SelectLabel>Options</SelectLabel>
            <SelectOption>Value</SelectOption>
            <SelectOption>Value</SelectOption>
            <SelectOption>Value</SelectOption>
          </SelectGroup>
        </SelectListBox>
      </SelectRoot>
    ));
    ```

  </CodeExample>
